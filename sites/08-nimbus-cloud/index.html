nimbus-quickstart.sh 
# Provision a compute instance in us-east-1
curl https://api.nimbuscloud.io/v2/instances \
  -H "Authorization: Bearer $NIMBUS_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "region": "us-east-1",
    "plan":   "nc-standard-4",
    "image":  "ubuntu-24.04",
    "label":  "web-prod-01"
  }'

# → 201 Created
# { "id": "inst_7Qh2kf9wZ", "status": "provisioning" }
// npm install @nimbuscloud/sdk
import { Nimbus } from "@nimbuscloud/sdk";

const nimbus = new Nimbus(process.env.NIMBUS_API_KEY);

const instance = await nimbus.instances.create({
  region: "us-east-1",
  plan:   "nc-standard-4",
  image:  "ubuntu-24.04",
  label:  "web-prod-01",
});

console.log(instance.id, instance.status);
# pip install nimbuscloud
import os
from nimbuscloud import Nimbus

nimbus = Nimbus(api_key=os.environ["NIMBUS_API_KEY"])

instance = nimbus.instances.create(
    region="us-east-1",
    plan="nc-standard-4",
    image="ubuntu-24.04",
    label="web-prod-01",
)

print(instance.id, instance.status)
